//콜백 함수 : 함수를 다른 함수의 인자로 전달하여 나중에 실행. 
//함수를 값처럼 전달하는 예시, 함수 표현식에 관한 예시
//매개변수가 3개 있는 함수, ask(question, yes, no)
/*
    question (string) : 질문 내용
    yes (func): 동의시"Yes"라고 답한 경우 실행되는 함수
    no (func): 거절시 "No"라고 답한 경우 실행되는 함수
*/
 
 
    function question(question, yes, no) {
        if (confirm(question)) yes() ;
        else no();
    }
 
    function ok () {
        alert("동의");
    }
 
    function reject () {
        alert("거절");
    }
 
    question("동의하시나요?",ok, reject);
 
 
 
 
 
    let age = prompt("나이를 알려주세요.", 18);
 
    let welcome;
 
    if (age < 18) {
 
        welcome = function() {
            alert("안녕!");
        };
 
    } else {
 
        welcome = function() {
            alert("안녕하세요!");
        };
 
    }
 
    welcome(); // 제대로 동작합니다.
 
 
//위를 아래와 같이 가능
 
    let age = prompt("나이를 알려주세요.", 18);
 
    let welcome = (age < 18) ?

    function() { alert("안녕하세요!"); };
 
    welcome(); // 제대로 동작합니다.
