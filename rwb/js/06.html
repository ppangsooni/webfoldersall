<script>
    //or --> || (하나라도 참이면 참)
    //앞이 거짓일 경우 뒤를 순차적으로 따짐 
 
    true || alert("not printed");
    false || alert("printed");
 
    let firstName = "";
    let lastName = "";
    let nickName = "바이올렛";
 
    alert( firstName || lastName || nickName || "익명"); // 바이올렛
 
    //&&이 ||보다 우선순위가 높음 
    //and---->&&    (하나라도 거짓이면 거짓) 
    //앞이 거짓이면 끝
 
    true && alert("not printed");
    false && alert("printed");
 
 
    //not ---> ! (qnwjd)
    //NOT을 두 개 연달아 사용(!!)하면 값을 불린형으로 변환할 수 있습니다.
 
 
 
    /* 정리 (정의)
        || (or) --> 하나라도 참이면 참 (참인 내용을 앞에 둠)
            예시     A || B --> A가 참이면 B는 생략
        
        && (and)  --> 하나라도 거짓이면 거짓 (거짓을 앞에 둠)
            예시     A && B --> A가 거짓이면 B는 생략
    */
 
    true || alert("true"); //console실행 안함
    false || alert("false");//console실행 함 
    true && alert("true"); //console실행 함
    false && alert("false");//console실행 안함 
</script>
