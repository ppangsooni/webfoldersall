<script>
 //문 1) if 와 문자열 0
 
        if ("0") {
        alert( 'Hello' );
        }
 
        //값 true 반환
    
        'use strict';
//문 2) 자바스크립트의 공식 이름
 
        //방법1 if문
        let javascriptName = prompt('자바스크립트의 공식이름은 무엇일까요?');
            if (javascriptName == "ECMAScript") {
                alert('정답입니다.') ;
            } else {
                alert('모르셨나요? 정답은 ECMAScript입니다!');
            }
 
        //방법2
        let javascriptName = (prompt('자바스크립트의 공식이름은 무엇일까요?') == 'ECMAScript') ? alert('정답입니다.') : alert('모르셨나요? 정답은 ECMAScript입니다!') ;
 
//문 3) 입력받은 숫자의 부호 표시하기
        
        //방법 1 if문
        let number = prompt('숫자를 입력하세요');
            if (number > 0) {
                alert("1");
            } else if (number < 0) {
                alert("-1");
            } else {
                alert("0");
            }
 
        //방법2
        let number = (prompt('숫자를 입력하세요') > 0) ? alert("1"):
        (number < 0) ? alert("-1"):
        alert("0");
 
//문 4) it문을 ?로 교체
 
        방법 1(문제) if 문
        let result;
            if (a + b < 4) {
            result = '미만';
            } else {
            result = '이상';
            }
 
        //방법 2 (답) ?
        let result = (a + b < 4) ? '미만' :'이상' ;
 
//문 5) if..else를 ?로 교체
 
        //방법 1(문제) if
 
        let message;
 
            if (login == '직원') {
            message = '안녕하세요.';
            } else if (login == '임원') {
            message = '환영합니다.';
            } else if (login == '') {
            message = '로그인이 필요합니다.';
            } else {
            message = '';
            }
 
        //방법 2(답) ?
        let message = (login == '직원') ? '안녕하세요.':
        (login == '임원') ? '환영합니다.':
        (login == '') ? '로그인이 필요합니다.':
        '';
</script>
