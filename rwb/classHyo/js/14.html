<script type="text/javascript">
    //참조
    //값에 의한 참조 : call by value => 원시 데이터
    //주소에 의한 참조 : call by referance => 객체
 
 
    let a = 10;
    let b = a; //값이 할당
 
    b++;
    console.log(a); // 10
    console.log(b); //11 이유는 값만으로 했기때문에
                //값에 의한 복사는 별개
    let aObj = {
        val: 10
    };
    let bObj = aObj;
    bObj.val++; //객체는 값을 복사시(즉, 주소를 넣는/주소에 의한 참조 )
                //객체에 의한 복사는 별개가 아닌 같이 
                //이는 두가지 얕은복사와 깊은 복사가 있다.
    console.dir(aObj);//11
    console.dir(bObj);//11
 
 
 
//중첩복사
let user = {
  name: "John",
  sizes: {
    height: 182,
    width: 50
  }
};
 
alert( user.sizes.height ); // 182
alert( user["sizes"][height]); // 182
alert( user["sizes"].height); // 182
 
 
//JSON 제이슨
let clone = JSON.parse(JSON.stringify(user)); //완전복사
 
//JSON.stringify => 객체(그냥자료)를 문자화 (직렬화)
//JSON.parse => 문자를 객체로
 
 
clone.sizes.height =192;
console.log(user);
console.log(clone);
</script>
